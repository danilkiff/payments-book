% kaobook-patches.tex — project-local patch layer поверх vendored kaobook 0.9.8.
% TeX Live / project-specific совместимость держим здесь, а не в forked class.

% Pre-load math fonts to avoid the TeX Live 2026 microtype/expl3 recursion
% that can be triggered when font config files are first loaded inside figures.
\AtBeginDocument{\setbox0=\hbox{$\alpha$}}

% PT Serif не даёт real small caps, поэтому для part-заголовков и их TOC-entry
% используем обычный полужирный upright-стиль без фолбэка в псевдо-small-caps.
\setkomafont{part}{\normalfont\bfseries}
\setkomafont{partentry}{\normalfont\bfseries}
