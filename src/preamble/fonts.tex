% fonts.tex — настройка шрифтов для XeLaTeX + русский язык

\usepackage{fontspec}

%% Основной шрифт (с засечками) — хорошо читается в печатном виде.
%% Требует установки: https://fonts.google.com/specimen/PT+Serif
\setmainfont{PT Serif}[
  Ligatures = TeX,
  Numbers   = OldStyle,
]

%% Шрифт без засечек — для подписей, колонтитулов, боковых заметок.
%% Требует установки: https://fonts.google.com/specimen/PT+Sans
\setsansfont{PT Sans}[
  Ligatures = TeX,
]

%% Моноширинный шрифт — для листингов кода, hex-дампов, идентификаторов.
%% PT Mono входит в стандартный набор macOS.
\setmonofont{PT Mono}[
  Scale = 0.88,
]

%% Явные объявления кириллических шрифтов для polyglossia.
\newfontfamily\cyrillicfont{PT Serif}[Ligatures=TeX, Script=Cyrillic]
\newfontfamily\cyrillicfontsf{PT Sans}[Ligatures=TeX, Script=Cyrillic]
\newfontfamily\cyrillicfonttt{PT Mono}[Scale=0.88]
