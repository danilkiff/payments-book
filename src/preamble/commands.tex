% commands.tex — кастомные макросы для книги о платёжном стеке

%% ── Платёжные термины ────────────────────────────────────────────────────────
% Единообразное форматирование специфических терминов по всей книге.

\DeclareRobustCommand{\codeword}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\domain}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\endpoint}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\filepath}[1]{\texorpdfstring{\path{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\iso}[1]{\texorpdfstring{\textsf{ISO~#1}}{ISO #1}}
\DeclareRobustCommand{\mti}[1]{\codeword{#1}}     % MTI значение
\DeclareRobustCommand{\de}[1]{DE~\codeword{#1}}   % Data Element: \de{55}
\DeclareRobustCommand{\rc}[1]{\codeword{#1}}      % Response Code: \rc{05}
\DeclareRobustCommand{\pan}[1]{\codeword{#1}}     % PAN/номер карты
\DeclareRobustCommand{\hex}[1]{\codeword{0x#1}}   % hex-значение
\newcommand{\law}[1]{\textit{#1}}                 % нормативный акт: \law{161-ФЗ}
\DeclareRobustCommand{\scheme}[1]{\texorpdfstring{\textsf{#1}}{#1}}

\pdfstringdefDisableCommands{%
  \def\codeword#1{#1}%
  \def\domain#1{#1}%
  \def\endpoint#1{#1}%
  \def\filepath#1{#1}%
  \def\mti#1{#1}%
  \def\de#1{DE #1}%
  \def\rc#1{#1}%
  \def\pan#1{#1}%
  \def\hex#1{0x#1}%
  \def\law#1{#1}%
}

%% ── Специальные блоки ────────────────────────────────────────────────────────
% Определения, предупреждения и справочные блоки через kaobook environments.

% "На практике" — практический совет
\newenvironment{practice}{%
  \begin{kaobox}[title=На практике]
}{%
  \end{kaobox}
}

% "Важно" — критическое замечание
\newenvironment{important}{%
  \begin{kaobox}[title=Важно, colback=red!5, colbacktitle=red!20]
}{%
  \end{kaobox}
}

% "Определение" — глоссарийная вставка внутри текста
\newenvironment{termdef}[1]{%
  \begin{kaobox}[title={Определение: #1}, colback=blue!4, colbacktitle=blue!15]
}{%
  \end{kaobox}
}

%% ── Todo-заметки ─────────────────────────────────────────────────────────────
% Цветовые категории для \todo{}:
\newcommand{\authoringnote}[1]{%
  \ifpaymentsbookrelease
    \relax
  \else
    #1%
  \fi
}
\newcommand{\todobox}[4]{%
  \begin{kaobox}[title={#1}, colback=#2, colbacktitle=#3]%
    #4%
  \end{kaobox}%
}
\newcommand{\todofig}[1]{\authoringnote{\todobox{Рисунок}{blue!8}{blue!35!black}{#1}}}
\newcommand{\todocite}[1]{\authoringnote{\todobox{Источник}{green!12}{green!35!black}{#1}}}
\newcommand{\todocheck}[1]{\authoringnote{\todobox{Проверить}{orange!18}{orange!55!black}{#1}}}
