\newtcolorbox{margincalloutbox}[2][]{%
  nobeforeafter,
  sharp corners,
  boxrule=0pt,
  titlerule=.4pt,
  left=4pt,
  right=4pt,
  top=2pt,
  bottom=2pt,
  lefttitle=4pt,
  righttitle=4pt,
  toptitle=2pt,
  bottomtitle=2pt,
  width=\linewidth,
  colback=RoyalBlue!25!White,
  colbacktitle=RoyalBlue!25!White,
  coltitle=black,
  colframe=black,
  fonttitle=\bfseries\footnotesize,
  before upper={\@margin@par},
  title={#2},
  #1
}

\NewDocumentCommand{\margincallout@place}{ o m m +m }{%
  \IfNoValueTF{#1}{%
    \marginnote{%
      \begin{minipage}{\marginparwidth}%
        \begin{margincalloutbox}[#2]{#3}%
          #4%
        \end{margincalloutbox}%
      \end{minipage}%
    }%
  }{%
    \marginnote[#1]{%
      \begin{minipage}{\marginparwidth}%
        \begin{margincalloutbox}[#2]{#3}%
          #4%
        \end{margincalloutbox}%
      \end{minipage}%
    }%
  }%
}

\NewDocumentCommand{\margindefinition}{ o m +m }{%

}

\newsavebox{\chaptersummarybox}
\newlength{\chaptersummaryfullwidth}
\newlength{\chaptersummaryneededspace}

\NewDocumentEnvironment{chaptersummary}{+b}{%
  \FloatBarrier
  \par
  \begingroup
  \setlength{\chaptersummaryfullwidth}{\dimexpr\textwidth+\marginparwidth+\marginparsep\relax}
  \sbox{\chaptersummarybox}{%
    \begin{minipage}{\chaptersummaryfullwidth}
      #1
    \end{minipage}%
  }%
  \setlength{\chaptersummaryneededspace}{%
    \dimexpr\ht\chaptersummarybox+\dp\chaptersummarybox+4\baselineskip\relax
  }%
  \ifdim\chaptersummaryneededspace>\textheight
    \section*{Итоги главы}
    #1
  \else
    \Needspace{\chaptersummaryneededspace}
    \begin{widepar}
    \section*{Итоги главы}
    \usebox{\chaptersummarybox}
    \end{widepar}
  \fi
  \par
  \endgroup
}{}
