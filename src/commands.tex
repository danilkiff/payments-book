% Small semantic helpers used throughout the book text.

\DeclareRobustCommand{\codeword}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\domain}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\endpoint}[1]{\texorpdfstring{\nolinkurl{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\filepath}[1]{\texorpdfstring{\path{#1}}{\detokenize{#1}}}
\DeclareRobustCommand{\iso}[1]{\texorpdfstring{\textsf{ISO #1}}{ISO #1}}
\DeclareRobustCommand{\mti}[1]{\codeword{#1}}
\DeclareRobustCommand{\de}[1]{DE \codeword{#1}}
\DeclareRobustCommand{\rc}[1]{\codeword{#1}}
\DeclareRobustCommand{\pan}[1]{\codeword{#1}}
\DeclareRobustCommand{\hex}[1]{\codeword{0x#1}}
\newcommand{\law}[1]{\textit{#1}}
\DeclareRobustCommand{\scheme}[1]{\texorpdfstring{\textsf{#1}}{#1}}
\DeclareRobustCommand{\idx}[2][]{%
  #2%
  \if\relax\detokenize{#1}\relax
    \index{#2}%
  \else
    \index{#1}%
  \fi
}
\DeclareRobustCommand{\idxdef}[2][]{%
  #2%
  \if\relax\detokenize{#1}\relax
    \index{#2|textbf}%
  \else
    \index{#1|textbf}%
  \fi
}
% Centralized cross-references live in the backmatter index file, so they need
% an immediate write to survive \printindex closing the named imakeidx stream.
\newcommand{\idximmediateentry}[1]{%
  \begingroup
  \expandafter\immediate\expandafter\write\csname\jobname @idxfile\endcsname{%
    \string\indexentry{#1}{\thepage}}%
  \endgroup
}
\newcommand{\idxsee}[2]{\idximmediateentry{#1|see{#2}}}
\newcommand{\idxseealso}[2]{\idximmediateentry{#1|seealso{#2}}}

  \par\smallskip
  {\footnotesize\noindent\fullcite{#1}\par}%
}
