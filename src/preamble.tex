% Пакеты LaTeX и глобальные настройки книги.
% Редактору: визуальные стили (цвета, notebox, маркер) — в src/styles.tex.

% --- Кодировки и язык ---
\usepackage[T2A,T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage[main=russian,english]{babel}

% --- Шрифты: PT Serif (текст), PT Sans (заголовки/блоки), PT Mono (код) ---
% Полная кириллица T2A для всех трёх гарнитур.
\usepackage{paratype}

% --- Поля для B5 ---
\usepackage[left=20mm,right=20mm,top=10mm,bottom=10mm,includeheadfoot]{geometry}

% --- Типографика и таблицы ---
\usepackage{amsmath}
\usepackage{graphicx}
\usepackage{tabularx}
\usepackage{ragged2e}
\usepackage{booktabs}
\usepackage[final]{microtype}
\usepackage{csquotes}
\usepackage{siunitx}
\sisetup{per-mode=symbol,output-decimal-marker={,}}
\usepackage{icomma}
\usepackage[font=small,labelfont=bf]{caption}
\DeclareCaptionLabelSeparator{emdash}{~--- }
\captionsetup{labelsep=emdash}
\usepackage[backend=biber,style=numeric-comp,sorting=nyt]{biblatex}
% Список источников: мелкий шрифт; URL и note — каждый с новой строки,
% note — серым.
\AtBeginBibliography{\footnotesize}
\DeclareFieldFormat{url}{\newline{\scriptsize\url{#1}}}
\DeclareFieldFormat{note}{\newline\textcolor{black!55}{#1}}
\usepackage[unicode]{hyperref}
\usepackage{enumitem}
\usepackage{needspace}
\usepackage{float}
\usepackage{xurl}
\usepackage{indentfirst}
\usepackage{eso-pic}
\usepackage{ccicons}
\usepackage{soul}

% --- Глоссарий аббревиатур ---
% glossaries-extra даёт \printunsrtglossary, печатающую записи в порядке
% объявления и не требующую внешнего makeglossaries-прохода.
\usepackage[nopostdot,nonumberlist,nogroupskip,toc=true,style=long]{glossaries-extra}

% --- Визуальные стили (цвета, tcolorbox, notebox-блоки, маркер) ---
\input{src/styles}

% --- Метаданные PDF ---
\hypersetup{
  pdftitle={Инженерия платежей: открытая книга о~карточных платежах, СБП и платёжной инфраструктуре},
  pdfauthor={},
  hidelinks,
}

% --- Шрифты в KOMA-элементах ---
% Заголовки/части/главы/секции — PT Sans, для визуального контраста с serif-телом.
\addtokomafont{disposition}{\sffamily}
\addtokomafont{pagehead}{\rmfamily\small}
\addtokomafont{caption}{\sffamily\small}
\addtokomafont{captionlabel}{\sffamily\bfseries\small}

% --- Часть: убрать разрыв страницы после заголовка части ---
\makeatletter
\renewcommand*{\@endpart}{\par\vspace{2\baselineskip}}
\makeatother

% --- Глобальные параметры набора ---
\raggedbottom
\extrafloats{200}
\setlength{\parindent}{1.25em}
\setlength{\parskip}{0pt}
\setlength{\emergencystretch}{1.5em}
\setlist[itemize,enumerate]{leftmargin=*,labelsep=0.55em,
itemsep=0pt,parsep=0pt,topsep=0pt,partopsep=0pt}

% --- Команда включения фигур ---
% \includefiguresvg[width=<dim>]{path-without-ext}  — Excalidraw-SVG → PDF
% Корпусные фигуры ограничиваем по высоте, чтобы они не доминировали на полосе.
\newcommand{\figuremaxheight}{.35\textheight}
\newcommand{\includefiguresvg}[2][width=\linewidth]{%
  \includegraphics[height=\figuremaxheight,keepaspectratio,#1]{#2.pdf}%
}

% --- \fullwidth: после удаления колонки полей равен \textwidth ---
% \AtBeginDocument нужен: KOMA финализирует \textwidth только в \begin{document}
\newlength{\fullwidth}
\AtBeginDocument{\setlength{\fullwidth}{\textwidth}}
