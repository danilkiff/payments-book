% Minimal project preamble kept close to stock kaobook.
% XeTeX in this environment does not resolve the TeX Live Libertinus family
% names used by kaobook, so we load the same font stack explicitly by file.

\usepackage{fontspec}
\usepackage{amssymb}
\usepackage{unicode-math}
\defaultfontfeatures{Ligatures=TeX}
\setmainfont{LibertinusSerif-Regular.otf}[
  BoldFont=LibertinusSerif-Bold.otf,
  ItalicFont=LibertinusSerif-Italic.otf,
  BoldItalicFont=LibertinusSerif-BoldItalic.otf,
  Scale=1.04,
]
\setsansfont{LibertinusSans-Regular.otf}[
  BoldFont=LibertinusSans-Bold.otf,
  ItalicFont=LibertinusSans-Italic.otf,
  BoldItalicFont=LibertinusSans-Bold.otf,
]
\setmonofont{LibertinusMono-Regular.otf}[
  BoldFont=LibertinusMono-Regular.otf,
  ItalicFont=LibertinusMono-Regular.otf,
  BoldItalicFont=LibertinusMono-Regular.otf,
  Scale=.89,
]
\setmathfont{LibertinusMath-Regular.otf}

\usepackage{polyglossia}
\setdefaultlanguage{russian}
\setotherlanguage{english}

\usepackage{newunicodechar}
\usepackage{longtable}
\usepackage{tabularx}
\usepackage{pgf-umlsd}
\usepackage{csquotes}
\usepackage{kaobiblio}
\usepackage{pdflscape}

\usetikzlibrary{arrows.meta, backgrounds, fit, positioning, shapes.geometric}

\hypersetup{
  pdftitle={Платёжный стек},
  pdfauthor={},
  colorlinks=true,
  linkcolor=RoyalBlue,
  citecolor=ForestGreen,
  urlcolor=Maroon,
}

\setcounter{margintocdepth}{1}
\setcounter{tocdepth}{1}
\renewcommand{\arraystretch}{1.25}
