% Пакеты LaTeX и глобальные настройки книги.
% Редактору: визуальные стили (цвета, notebox, маркер) — в src/styles.tex.

% --- Кодировки и язык ---
\usepackage[T2A,T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage[main=russian,english]{babel}

% --- Шрифты: PT Serif (текст), PT Sans (заголовки/блоки), PT Mono (код) ---
% Полная кириллица T2A для всех трёх гарнитур.
\usepackage{paratype}

% --- Поля для B5 ---
\usepackage[left=10mm,right=10mm,top=5mm,bottom=5mm,includeheadfoot]{geometry}

% --- Типографика и таблицы ---
\usepackage{amsmath}
\usepackage{graphicx}
\usepackage{tabularx}
\usepackage{longtable}
\usepackage{float}
% Сквозная нумерация таблиц и~рисунков с~единицы по всей книге
% (вместо дефолтной KOMA-нумерации «глава.номер»);
% \counterwithout~--- ядро LaTeX2e, отдельный пакет chngcntr не~нужен.
\counterwithout{table}{chapter}
\counterwithout{figure}{chapter}
% Нумерация сносок с единицы на каждой странице.
\usepackage{perpage}
\MakePerPage{footnote}
\usepackage{ragged2e}
\usepackage{booktabs}
\usepackage[final]{microtype}
\usepackage{csquotes}
\usepackage[font=small,labelfont=bf]{caption}
\DeclareCaptionLabelSeparator{emdash}{~--- }
\captionsetup{labelsep=emdash}
\usepackage[backend=biber,style=numeric-comp,sorting=none]{biblatex}
\usepackage[unicode]{hyperref}
% Список источников: мелкий шрифт; URL не печатается отдельной строкой,
% title кликабельный (href на url); note — серым с новой строки.
\AtBeginBibliography{\footnotesize\setlength{\bibitemsep}{0pt plus 0.3ex}}
\DeclareFieldFormat{url}{}
\DeclareFieldFormat{title}{%
  \iffieldundef{url}{\mkbibemph{#1}}{\href{\thefield{url}}{\mkbibemph{#1}}}%
}
\DeclareFieldFormat[article,inproceedings,inbook,incollection,thesis,patent,unpublished]{title}{%
  \iffieldundef{url}{\mkbibquote{#1\isdot}}{\href{\thefield{url}}{\mkbibquote{#1\isdot}}}%
}
\DeclareFieldFormat{note}{}
% Записи с keyword=noauthor печатаются без author и без year
% (для ФЗ — снимает тавтологии «Федеральное Собрание РФ. Федеральный
% закон от...» и «в силе с DD.MM.YYYY. YYYY.», обе даты уже в title).
\AtEveryBibitem{\ifkeyword{noauthor}{%
  \clearname{author}\clearfield{year}\clearfield{date}%
}{}}
\usepackage{bookmark}
\usepackage{enumitem}
\usepackage{xurl}
\usepackage{indentfirst}
\usepackage{eso-pic}
\usepackage{ccicons}
\usepackage{soul}
\usepackage{needspace}
\usepackage{placeins}

% --- Визуальные стили (цвета, tcolorbox, notebox-блоки, маркер) ---
\input{src/styles}

% --- Метаданные PDF ---
\hypersetup{
  pdftitle={Инженерия платежей: открытая книга о~карточных платежах, СБП и платёжной инфраструктуре},
  pdfauthor={},
  hidelinks,
}

% --- Шрифты в KOMA-элементах ---
% Заголовки/части/главы/секции — PT Sans, для визуального контраста с serif-телом.
\addtokomafont{disposition}{\sffamily}
\addtokomafont{pagehead}{\rmfamily\small}
\addtokomafont{caption}{\sffamily\small}
\addtokomafont{captionlabel}{\sffamily\bfseries\small}

% --- Часть: убрать разрыв страницы после заголовка части ---
\makeatletter
\renewcommand*{\@endpart}{\par\vspace{2\baselineskip}}
\makeatother

% --- Ширина номеров в оглавлении ---
% Дефолтный numwidth у section в KOMA рассчитан под "X.Y" и не вмещает
% двухзначные подсекции вида "18.10"/"18.11" — номер наезжает на название.
\DeclareTOCStyleEntry[numwidth=3.2em]{default}{section}

% --- Уплотнение TOC: убрать вертикальную паузу перед каждой главой ---
% KOMA по умолчанию ставит ~1em перед записью \chapter в TOC; на 38 главах
% это съедает ~0.7 страницы. Глава по-прежнему различима по жирности и
% номеру слева.
\RedeclareSectionCommand[tocbeforeskip=0.2em]{chapter}

% --- Глобальные параметры набора ---
\raggedbottom
\extrafloats{200}
% Параметры размещения floats: пускаем таблицы свободнее в общий поток,
% чтобы не уезжали в одиночную «страницу-сирота». Float-page разрешён,
% только если заполняет ≥85% полосы.
\renewcommand{\textfraction}{0.10}
\renewcommand{\topfraction}{0.90}
\renewcommand{\bottomfraction}{0.70}
\renewcommand{\floatpagefraction}{0.85}
\setlength{\parindent}{1.25em}
\setlength{\parskip}{0pt}
\setlength{\emergencystretch}{1.5em}
\setlist[itemize,enumerate]{leftmargin=*,labelindent=\parindent,
labelsep=0.55em,itemsep=0pt,parsep=0pt,topsep=0pt,partopsep=0pt}

% --- Команда включения фигур ---
% \includefiguresvg[width=<dim>]{path-without-ext}  — Inkscape SVG → PDF
% Корпусные фигуры ограничиваем по высоте, чтобы они не доминировали на полосе.
\newcommand{\figuremaxheight}{.45\textheight}
\newcommand{\includefiguresvg}[2][width=\linewidth]{%
  \includegraphics[height=\figuremaxheight,keepaspectratio,#1]{#2.pdf}%
}
