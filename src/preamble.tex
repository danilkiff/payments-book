\newcommand{\booknotebox}[1]{%
  \begingroup
  \setlength{\fboxsep}{5pt}%
  \colorbox{Soft}{%
    \parbox[t]{\dimexpr\marginparwidth-2\fboxsep\relax}{%
      \sffamily
      \footnotesize
      \RaggedRight
      \setlength{\parindent}{0pt}%
      \setlength{\parskip}{3pt}%
      #1%
    }%
  }%
  \endgroup
}

\newcommand{\booknote}[2][0pt]{%
  \leavevmode\marginpar{\vspace*{#1}\booknotebox{#2}}%
}

\newcommand{\definitionnote}[3][0pt]{%
  \booknote[#1]{\textbf{#2}\par #3}%
}

\newcommand{\marginfigure}[3][0pt]{%
  \booknote[#1]{%
    \centering
    #2\par
    \smallskip
    \RaggedRight
    #3%
  }%
}

\newcommand{\sidefootnote}[2][0pt]{%
  \refstepcounter{footnote}%
  \textsuperscript{\thefootnote}%
  \booknote[#1]{\textsuperscript{\thefootnote}\enspace #2}%
}
