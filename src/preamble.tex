\newcommand{\booknotebox}[1]{%
  \begingroup
  \setlength{\fboxsep}{4pt}%
  \setlength{\fboxrule}{0.4pt}%
  \fbox{%
    \parbox[t]{\dimexpr\marginparwidth-2\fboxsep-2\fboxrule\relax}{%
      \sffamily
      \scriptsize
      \RaggedRight
      \setlength{\parindent}{0pt}%
      \setlength{\parskip}{2pt}%
      #1%
    }%
  }%
  \endgroup
}

\newcommand{\colormarginalnote}[5][0pt]{%
  \leavevmode\marginpar{%
    \vspace*{#1}%
    \parbox[t]{\marginparwidth}{%
      \sffamily
      \scriptsize
      \RaggedRight
      \setlength{\parindent}{0pt}%
      \setlength{\parskip}{2.5pt}%
      \color{#5}%
      \textbf{#2}%
      \if\relax\detokenize{#3}\relax
        \par
      \else
        \par{\bfseries\footnotesize #3}\par
      \fi
      #4%
    }%
  }%
}

\newcommand{\booknote}[2][0pt]{%
  \leavevmode\marginpar{\vspace*{#1}\booknotebox{#2}}%
}

\newcommand{\marginalnote}[4][0pt]{%
  \colormarginalnote[#1]{#2}{}{#3}{#4}%
}

\newcommand{\definitionnote}[3][0pt]{%
  \colormarginalnote[#1]{Определение}{#2}{#3}{DefinitionInk}%
}

\newcommand{\importantnote}[2][0pt]{%

\newcommand{\practicenote}[2][0pt]{%

\newcommand{\marginfigure}[3][0pt]{%

  }%
