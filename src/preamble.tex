% DOI как обычный текст (PT Serif), а не моноширинный \nolinkurl: дефолтный
% формат biblatex печатает DOI через \nolinkurl, который под xurl рвётся
% после любого символа — год «1989» уезжал как «198»+«9». Печатаем DOI
% телом текста и разрешаем перенос только после «.» и «/», поэтому
% числовые группы (1989, 10478785) не разрываются.
\makeatletter
\newcommand*{\pmt@doitext}[1]{\pmt@doitext@scan#1\@nil}
\def\pmt@doitext@scan#1{%
  \ifx#1\@nil\else
  #1%
  \ifx#1.\allowbreak\else\ifx#1/\allowbreak\fi\fi
  \expandafter\pmt@doitext@scan
\fi}
\DeclareFieldFormat{doi}{%
  \mkbibacro{DOI}\addcolon\space
  \ifhyperref
  {\href{https://doi.org/#1}{\pmt@doitext{#1}}}
{\pmt@doitext{#1}}}
\makeatother

% --- Ширина номеров в оглавлении ---
% Дефолтный numwidth у section в KOMA рассчитан под "X.Y" и не вмещает
% двухзначные подсекции вида "18.10"/"18.11" — номер наезжает на название.
\DeclareTOCStyleEntry[numwidth=3.2em]{default}{section}

% --- Уплотнение TOC: убрать вертикальную паузу перед каждой главой ---
% KOMA по умолчанию ставит ~1em перед записью \chapter в TOC; на 38 главах
% это съедает ~0.7 страницы. Глава по-прежнему различима по жирности и
% номеру слева.
\RedeclareSectionCommand[tocbeforeskip=0.2em]{chapter}

% --- Заголовок каждой главы прижат к верхней линии полосы ---
% KOMA по умолчанию ставит \vspace*{2.3\baselineskip} над \chapter/\chapter*
% (через \chapterheadstartvskip) — классический «book-opener» приём, треть
% полосы белым полем. На reference-книге это лишняя «парадная» пауза;
% обнуляем, чтобы любое название (TOC, фронт-/бэк-матёр, главы 1–37)
% начиналось у верхней линии полосы.
\renewcommand*{\chapterheadstartvskip}{\vspace*{0pt}}

% --- Часть в TOC: «Часть N. Название» и визуальное отделение ---
% Голая римская цифра «I» в первой колонке читается как номер главы или
% подсекции; русская издательская норма (Мильчин) требует слова «Часть».
% Заодно: beforeskip 1.5em над частью даёт макроструктурный отрыв от
% предыдущей главы (1.5× межсекционного шага); вес шрифта KOMA-default
% для part уже \large\bfseries — не трогаем, иерархия и так появится.
\newcommand*{\tocpartnumberformat}[1]{Часть~#1}
\DeclareTOCStyleEntry[
  entrynumberformat=\tocpartnumberformat,
  numwidth=5.5em,
  beforeskip=1.5em plus 0.2em minus 0.1em,
]{tocline}{part}
