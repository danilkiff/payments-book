% Keep the vendored kaobook files pristine; project-specific fixes live here.

% Disable kaobook's caption-rewriting environment hooks.
% They recurse on recent TeX Live releases and are not worth preserving here.
\csdef{@begin@figure@hook}{}
\csdef{@end@figure@hook}{}
\csdef{@begin@table@hook}{}
\csdef{@end@table@hook}{}

% Use the margin for kaobook navigation, not duplicated bibliography cards.
\AfterEndPreamble{%
  \RenewDocumentCommand{\sidecite}{o o o m}{%
    \IfNoValueOrEmptyTF{#2}%
      {\cite{#4}}%
      {\IfNoValueOrEmptyTF{#3}%
        {\IfNoValueTF{#3}{\cite[#2]{#4}}{\cite[#2][]{#4}}}%
        {\cite[#2][#3]{#4}}}%
  }%
  \RenewDocumentCommand{\sidesupercite}{o o o m}{%
    \IfNoValueOrEmptyTF{#2}%
      {\supercite{#4}}%
      {\IfNoValueOrEmptyTF{#3}%
        {\IfNoValueTF{#3}{\supercite[#2]{#4}}{\supercite[#2][]{#4}}}%
        {\supercite[#2][#3]{#4}}}%
  }%
  \RenewDocumentCommand{\sidetextcite}{o o o m}{%
    \IfNoValueOrEmptyTF{#2}%
      {\textcite{#4}}%
      {\IfNoValueOrEmptyTF{#3}%
        {\IfNoValueTF{#3}{\textcite[#2]{#4}}{\textcite[#2][]{#4}}}%
        {\textcite[#2][#3]{#4}}}%
  }%
  \RenewDocumentCommand{\sideparencite}{s o o o m}{%
    \IfBooleanTF{#1}%
      {\IfNoValueOrEmptyTF{#3}%
        {\parencite*{#5}}%
        {\IfNoValueOrEmptyTF{#4}%
          {\IfNoValueTF{#4}{\parencite*[#3]{#5}}{\parencite*[#3][]{#5}}}%
          {\parencite*[#3][#4]{#5}}}}%
      {\IfNoValueOrEmptyTF{#3}%
        {\parencite{#5}}%
        {\IfNoValueOrEmptyTF{#4}%
          {\IfNoValueTF{#4}{\parencite[#3]{#5}}{\parencite[#3][]{#5}}}%
          {\parencite[#3][#4]{#5}}}}%
  }%
}
